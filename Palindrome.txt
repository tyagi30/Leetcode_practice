

Palindrome problem()
time complexity =O(n)
Space = O(1)



class Solution {
    public boolean isPalindrome(int x) {
        
int rev=0;
int y=x;

        while(x>0)
        {   
           rev=rev*10+x%10;
                  
            x=x/10;
        

        }
        if(y==rev)
        {return true;
        }
        else 
        {return false;
    }
    }
}
